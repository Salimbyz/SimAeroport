43 static vector<int> genererRetardsPoisson(int moyenne, int nombre, double probaSansRetard) {
44 random_device aleatoire;
45 mt19937 gen(aleatoire());
46
47 poisson_distribution<> d(moyenne);
48 uniform_real_distribution<> proba(0.0, 1.0);
49
50 std::vector<int> retards;
51 for (int n = 0; n < nombre; ++n) {
52 if (proba(gen) > probaSansRetard) {
53 // Générer un retard
54 retards.push_back(d(gen));
